\[
F^{\left( k \right)}\left( s,t \right) \equiv \left< \exp \left( se^{-i\varphi} \right) \right> ^{\left( k \right)}=\left< \sum_{m=0}^{\infty}{\frac{s^m}{m!}e^{-im\varphi}} \right> ^{\left( k \right)},

\[

\[

\[
\varPsi ^{\left( k \right)}\left( s,t \right) \equiv s\frac{\partial}{\partial s}\ln F^{\left( k \right)}\left( s,t \right) =\frac{s}{F^{\left( k \right)}}\frac{\partial}{\partial s}F^{\left( k \right)}\equiv \sum_{m=1}^{\infty}{\varkappa _{m}^{\left( k \right)}\left( t \right) s^m}, \varkappa _{m}^{\left( k \right)}\left( t \right) =\frac{1}{m!}\frac{\partial ^m}{\partial s^m}\left. \varPsi ^{\left( k \right)}\left( s,t \right) \right|_{s=0}.
\]
\[
\text{简化记号：}F\text{默认为}F^{\left( k \right)}\text{，}F^{\left( j \right)}\text{总是保留，}A=i\omega _0+\gamma , B=h_k, C={h_k}^*, D=r_{j\rightarrow k}\frac{m_j}{m_k},

\[
\partial _{t}F=-As\partial _sF-Bs\partial _{s}^{2}F+CsF+D\left( F^{\left( j \right)}-F \right) ,
\]
\[
{\color[RGB]{192, 192, 192} \text{交换求导顺序大大化简计算：}\partial _t\varPsi =s\partial _t\left( \frac{\partial _sF}{F} \right) =\frac{s}{F}\partial _t\partial _sF-\frac{s}{F^2}\partial _tF\partial _sF=s\partial _s\left( \frac{\partial _tF}{F} \right) ,}

\[
{\color[RGB]{192, 192, 192} 1: \partial _sF=\frac{\varPsi}{s}F, 2: \partial _{s}^{2}F=\partial _s\left( \frac{\varPsi}{s}F \right) =\frac{\partial _s\varPsi s-\varPsi}{s^2}F+\frac{\varPsi}{s}\frac{\varPsi}{s}F=\frac{F}{s^2}\left( s\partial _s\varPsi -\varPsi +\varPsi ^2 \right) }

\[

\[
\#: \partial _t\varPsi =-As\partial _s\varPsi -Bs\partial _s\left( \partial _s\varPsi -\frac{\varPsi}{s}+\frac{\varPsi ^2}{s} \right) +Cs+Ds\partial _s\left( \frac{F^{\left( j \right)}}{F^{\left( k \right)}} \right) .

\[
{\color[RGB]{192, 192, 192} \partial _s\left( \frac{F^{\left( j \right)}}{F^{\left( k \right)}} \right) =\frac{\frac{\varPsi ^{\left( j \right)}F^{\left( j \right)}}{s}F^{\left( k \right)}-F^{\left( j \right)}\frac{\varPsi ^{\left( k \right)}F^{\left( k \right)}}{s}}{\left( F^{\left( k \right)} \right) ^2}=\frac{1}{s}\frac{F^{\left( j \right)}}{F^{\left( k \right)}}\left( \varPsi ^{\left( j \right)}-\varPsi ^{\left( k \right)} \right) , \partial _s\varPsi -\frac{\varPsi}{s}=s\partial _s\left( \frac{\varPsi}{s} \right) }

\[
\#: \partial _t\varPsi =-As\partial _s\varPsi -Bs\partial _s\left( s\partial _s\left( \frac{\varPsi}{s} \right) +\frac{\varPsi ^2}{s} \right) +Cs+D\frac{F^{\left( j \right)}}{F^{\left( k \right)}}\left( \varPsi ^{\left( j \right)}-\varPsi ^{\left( k \right)} \right) .

\[
\text{方程代入级数匹配}\sum_{m=1}^{\infty}{c_n}s^m\text{形式的系数}c_n:

\[
\varPsi ^{\left( k \right)}=\sum_{m=1}^{\infty}{\varkappa _{m}^{\left( k \right)}s^m}, F^{\left( k \right)}=\sum_{m=0}^{\infty}{A_{m}^{\left( k \right)}\frac{s^m}{m!}},
\]
\[
\frac{\varPsi}{s}=\frac{\partial}{\partial s}\ln F, F\left( 0,t \right) =1, \ln F\left( 0,t \right) =0,
\]
\[
\ln F(\sigma ,t)\mid_{\sigma =0}^{\sigma =s}=\int_0^s{\frac{\Psi (\sigma ,t)}{\sigma}d\sigma}, \Psi (\sigma ,t)=\sum_{m=1}^{\infty}{\varkappa _m\sigma ^m},
\]
\[
\ln F(s,t)=\int_0^s{\sum_{m=1}^{\infty}{\varkappa _m\sigma ^{m-1}d\sigma}}=\sum_{m=1}^{\infty}{\frac{\varkappa _m}{m}s^m},
\]
\[
F(s,t)=\exp \left( \sum_{m=1}^{\infty}{\frac{\varkappa _m}{m}s^m} \right) , \Delta \varkappa _m=\varkappa _{m}^{(j)}-\varkappa _{m}^{(k)}, \Delta \varkappa _0=0
\]
\[
\varPsi ^{\left( j \right)}-\varPsi ^{\left( k \right)}\equiv \Delta V=\sum_{m=1}^{\infty}{\Delta \varkappa _ms^m}=\sum_{m=0}^{\infty}{\Delta \varkappa _ms^m},
\]
\[
\frac{F^{\left( j \right)}}{F^{\left( k \right)}}\equiv R\left( s,t \right) \equiv \sum_{m=0}^{\infty}{\rho _ms^m}=\exp \left( \sum_{m=0}^{\infty}{\frac{\Delta \varkappa _m}{m}s^m} \right) \equiv \mathrm{e}^{H\left( s,t \right)}.
\]
\[
s\frac{\partial}{\partial s}R=sR\frac{\partial}{\partial s}H=R\sum_{m=0}^{\infty}{\Delta \varkappa _ms^m}=R\Delta V,
\]
\[
{\color[RGB]{192, 192, 192}s\frac{\partial}{\partial s}R=\,\,\sum_{m=1}^{\infty}{m\rho _ms^m}, R\Delta V=\left( \sum_{i=0}^{\infty}{\rho _is^i} \right) \left( \sum_{j=0}^{\infty}{\Delta \varkappa _js^j} \right) =\sum_{n=1}^{\infty}{\left( \sum_{k=0}^{n-1}{\rho _k}\Delta \varkappa _{n-k} \right) s^n},}
\]
\[
m\rho _n=\sum_{k=0}^{m-1}{\rho _k}\Delta \varkappa _{m-k}, m-k=q,
\]
\[
\text{通过让}s=0\text{提取出}m=0\text{的常数项}: \rho _0=\frac{A_{0}^{\left( j \right)}}{A_{0}^{\left( k \right)}}=1,
\]
\[
s\partial _s\left( \frac{F^{\left( j \right)}}{F^{\left( k \right)}} \right) =\frac{F^{\left( j \right)}}{F^{\left( k \right)}}\left( \varPsi ^{\left( j \right)}-\varPsi ^{\left( k \right)} \right) =R\Delta V=s\frac{\partial}{\partial s}R=\sum_{m=1}^{\infty}{\left( m\rho _m \right) s^m}.
